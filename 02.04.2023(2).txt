#include <iostream>
#include <string>

using namespace std;

int main() {
    setlocale(LC_ALL, "Rus");
    string text;
    int letterCount = 0;

    cout << "Введите текст: ";
    getline(cin, text);

   
    for (unsigned int i = 0; i < text.length(); i++) {
        if (isalpha(text[i])) {
            letterCount++;
        }
    }

    cout << "Текст содержит : " << letterCount <<  " Букв" << endl;

    return 0;
}

#include <iostream>
#include <string>

using namespace std;

int main() {
    setlocale(LC_ALL, "Rus");
    string text;
    int digitCount = 0;

    cout << "Введите текст: ";
    getline(cin, text);

   
    for (unsigned int i = 0; i < text.length(); i++) {
        if (isdigit(text[i])) {
            digitCount++;
        }
    }

    cout << "Текст содержит " << digitCount << " Цифра" << endl;

    return 0;
}
