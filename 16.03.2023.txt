#include <iostream>
using namespace std;

void num(int* arr, int size, int* summ, int* mult)
{
    for (int i = 0; i < size; i++)
    {
        *summ += *(arr + i);
    }
    cout << "Sum: " << *summ << endl;
    for (int j = 0; j < size; j++)
    {
        *mult *= (*arr + j);
    }
    cout << "Mult: " << *mult << endl;
}

int main()
{

    const int size = 2;
    int arr[size];
    int summ = 0, multi = 1;
    int* numsumm = &summ;
    int* nummulti = &multi;
    for (int i = 0; i < size; i++)
    {
        arr[i] = rand() % 100;
    }
    int* pa = &arr[0];
    num(pa, size, numsumm, nummulti);

    return 0;
}
