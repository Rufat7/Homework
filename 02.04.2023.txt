#include <iostream>
using namespace std;
int main()
{
    setlocale(LC_ALL, "Rus");
    int num1 = 0, num2 = 0, num3 = 0;
    char str[100];
    cout << "Введите строку:" << endl;
    cin.getline(str, 500);
    for (int i = 0; i < strlen(str); i++)
    {
        if (isalpha(str[i]))
            num1++;
        else if (isdigit(str[i]))
            num2++;
        else
            num3++;
    }
    cout << "Количество букв в строке: " << num1 << endl;
    cout << "Количество цифр в строке: " << num2 << endl;
    cout << "Количество других символов в строке: " << num3 << endl;
    
return 0;
}

#include <iostream>
using namespace std;


bool num(string str)
{
    int num1 = 0;
    int num2 = str.length() - 1;

    while (num1 < num2)
    {
      
        if (str[num1] != str[num2]) {
            return false;
        }

        num1++;
        num2--;
    }

    return true;
}

int main()
{
    setlocale(LC_ALL, "Rus");
    string str = "1";
    cout << "Введите:";
    cin >> str;

    if (num(str)) {
        cout << "Palindrome";
    }
    else {
        cout << "Not Palindrome";
    }

    return 0;
}
